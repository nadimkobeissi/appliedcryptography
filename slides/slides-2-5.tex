\section{Introduction}

\begin{frame}{Securely implementing cryptography}
	\begin{itemize}
		\item Designing, implementing, and deploying cryptographic mechanisms is notoriously hard
		\item Regular discoveries of high-profile design flaws, devastating implementation bugs, and side-channel vulnerabilities
		\item Even in \textbf{widely deployed} mechanisms!
	\end{itemize}
	\vspace{1em}
	\begin{center}
		\Large
		Three levels of complexity:
		\begin{enumerate}
			\item \textbf{Design Level}: Security proofs, adversarial models
			\item \textbf{Implementation Level}: Correctness, performance, memory safety
			\item \textbf{Deployment Level}: Side-channel resistance
		\end{enumerate}
	\end{center}
\end{frame}

\begin{frame}{The problem}
	\begin{center}
		\Large
		Attackers can break designs, exploit bugs, recover secrets via side-channels, or \textbf{any combination}
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item \textbf{Pen-and-paper proofs}: Consider only ``cores,'' still error-prone
		\item \textbf{Aggressive optimization}: Increases bug risk
		\item \textbf{Current modus operandi}: Select few experts with rudimentary tools
	\end{itemize}
	\vspace{1em}
	\begin{center}
		This approach \textbf{cannot keep pace} with the rate of innovation in cryptography
	\end{center}
\end{frame}

\begin{frame}{Enter: Computer-Aided Cryptography (CAC)}
	\begin{itemize}
		\item Active area of research: \textbf{formal, machine-checkable} approaches to designing, analyzing, and implementing cryptography
		\item \textbf{Design level}: Tools manage complexity of security proofs, can reveal subtle flaws
		\item \textbf{Implementation level}: Guarantee correct behavior, even for highly optimized code
		\item \textbf{Deployment level}: Check side-channel protections
	\end{itemize}
	\vspace{1em}
	\begin{center}
		Individual tools may only address \textit{part} of the problem, but when \textbf{combined}, they provide a \textbf{high degree of assurance}
	\end{center}
\end{frame}

\begin{frame}{Success stories}
	\begin{itemize}
		\item Computer-aided security analyses were influential in \textbf{TLS 1.3} standardization
		\item \textbf{\haclstar library}: Integrated into Mozilla Firefox's NSS
		\item \textbf{Fiat Cryptography}: Elliptic curve code in Google's BoringSSL
		\item \textbf{EverCrypt}: Routines used in Linux kernel's Zinc crypto library
	\end{itemize}
	\begin{center}
		\textbf{Formally verified code at Internet scale!}
	\end{center}
\end{frame}

\begin{frame}{Missing: Real-world software complexity}
	\begin{itemize}
		\item Pen-and-paper proofs consider \textbf{isolated algorithms}
		\item Real deployments involve massive software stacks:
		      \begin{itemize}
			      \item Operating system layers
			      \item Network protocol implementations
			      \item Library dependencies
			      \item Hardware abstraction layers
		      \end{itemize}
		\item Each layer introduces new attack surfaces and unexpected interactions
		\item \textbf{Result}: Security proofs may not hold in practice!
	\end{itemize}
\end{frame}

\begin{frame}{Challenges ahead}
	\begin{itemize}
		\item CAC risks becoming a \textbf{victim of its own success}
		\item Trust can be undermined by difficulty understanding guarantees and fine-print caveats
		\item Field is increasingly broad, complex, and rapidly evolving
		\item \textbf{Post-quantum transition}: New algorithms need verification
		\item \textbf{Scaling}: Moving from primitives to whole cryptographic systems
	\end{itemize}
	\vspace{1em}
	\begin{center}
		\textit{No one has complete understanding of every facet}
	\end{center}
\end{frame}

\section{Design-Level Security}

\begin{frame}{Symbolic security: Example}
	\textbf{Symmetric encryption modeled by:}
	\begin{itemize}
		\item Functions: \texttt{Enc} and \texttt{Dec}
		\item Equational theory:
	\end{itemize}
	\begin{center}
		\Large
		\texttt{Dec(Enc(m,k),k) = m}
	\end{center}
	\begin{itemize}
		\item Decrypting ciphertext \texttt{Enc(m,k)} with key \texttt{k} recovers plaintext \texttt{m}
		\item Adversary can only use specified primitives and equations
		\item Can decrypt only if it has the \textbf{entire} secret key
	\end{itemize}
\end{frame}

\begin{frame}{Two flavors of design-level security}
	\begin{itemize}
		\item Developed over years in \textbf{two separate communities}:
		      \vspace{0.5em}
		\item \textbf{Symbolic Security} (Formal methods community)
		      \begin{itemize}
			      \item Messages are abstract \textbf{symbols}, not bits
			      \item Cryptographic primitives are \textbf{perfect black boxes}
			      \item Can only decrypt with the \textbf{exact} key
		      \end{itemize}
		      \vspace{0.5em}
		\item \textbf{Computational Security} (Cryptography community)
		      \begin{itemize}
			      \item Messages are \textbf{bitstrings} of specific length
			      \item Encryption is a \textbf{probabilistic algorithm}
			      \item Adversaries have \textbf{bounded computational power}
		      \end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Symbolic vs. Computational: Key differences}
	\begin{center}
		\begin{tabular}{|l|l|l|}
			\hline
			\textbf{Aspect} & \textbf{Symbolic} & \textbf{Computational}  \\
			\hline
			Messages        & Abstract terms    & Bitstrings              \\
			\hline
			Adversary       & Rule-based        & Probabilistic algorithm \\
			\hline
			Security        & All-or-nothing    & Probabilistic bounds    \\
			\hline
			Analysis        & Often decidable   & Generally undecidable   \\
			\hline
			Guarantees      & ``Cannot break''  & ``Hard to break''       \\
			\hline
		\end{tabular}
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item \textbf{Symbolic}: Quick protocol flaw detection, automated analysis
		\item \textbf{Computational}: Concrete security guarantees, cryptographic proofs
		\item \textbf{Both}: Different insights, complementary verification
	\end{itemize}
\end{frame}

\begin{frame}{Symbolic security properties}
	Two main flavors:
	\vspace{0.5em}
	\begin{enumerate}
		\item \textbf{Trace properties}
		      \begin{itemize}
			      \item Bad events never occur on any execution trace
			      \item Example: Secret data never in adversarial knowledge
		      \end{itemize}
		      \vspace{0.5em}
		\item \textbf{Equivalence properties}
		      \begin{itemize}
			      \item Adversary cannot distinguish between two protocols
			      \item Example: Cannot differentiate real secret from random value
		      \end{itemize}
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{Equivalence properties cannot be naturally expressed as trace properties}
	\end{center}
\end{frame}

\begin{frame}{Computational security properties}
	Characterized along two axes:
	\vspace{0.5em}
	\begin{center}
		\begin{tabular}{|c|c|}
			\hline
			\textbf{Type}    & \textbf{Quantification} \\
			\hline
			Game-based       & Concrete                \\
			Simulation-based & Asymptotic              \\
			\hline
		\end{tabular}
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item CAC tools primarily focus on \textbf{game-based, concrete security}
		\item Simulation-based proofs: More complex, support composition
		\item Concrete security implies asymptotic security
	\end{itemize}
\end{frame}

\begin{frame}{Why design-level security matters}
	\begin{itemize}
		\item Mathematical arguments are the \textbf{only way} to demonstrate security against entire classes of attacks
		\item Security proofs are \textbf{necessary} for any new standard
	\end{itemize}
	\vspace{1em}
	\textbf{But pen-and-paper arguments are alarmingly fragile:}
	\begin{enumerate}
		\item \textbf{Erroneous arguments}: Tedious, error-prone, errors go unnoticed for years
		\item \textbf{Inappropriate modeling}: Simplified models miss real attacks
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{CAC tools can detect flaws and manage complexity beyond pen-and-paper reach}
	\end{center}
\end{frame}

\begin{frame}{Key features of symbolic analysis tools}
	\begin{itemize}
		\item \textbf{Unbounded sessions}: Can analyze arbitrary number of concurrent sessions, proving \textbf{absence of attacks} in the model
		\item \textbf{Trace properties}: Bad events never occur on any execution trace
		\item \textbf{Equivalence properties}: Adversary cannot distinguish between two protocols
		\item \textbf{Equational theories}: Model algebraic properties of crypto operations (XOR, DH, etc.)
		\item \textbf{Global mutable state}: Support for shared databases, key servers
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		Modern unbounded tools often \textbf{outperform} bounded tools, even for small session counts!
	\end{center}
\end{frame}

\begin{frame}{Game-based security}
	\begin{itemize}
		\item Probabilistic experiment: ``game'' between challenger and adversary
		\item Security statement:
		      \begin{center}
			      \textit{``For all adversaries, probability of achieving goal $\leq$\ threshold''}
		      \end{center}
		\item Core methodology: \textbf{Game hopping}
		      \begin{enumerate}
			      \item Start with original game
			      \item Transform step-by-step
			      \item Reach game where success probability is known
			      \item Bound probability increases at each step
			      \item Similar to what we saw in Part 1 of this course!
		      \end{enumerate}
	\end{itemize}
\end{frame}

\begin{frame}{Key features of computational analysis tools}
	\textbf{Automation vs. Expressivity:} Every tool makes this tradeoff
	\begin{itemize}
		\item Automated tools: Push-button proofs, limited patterns
		\item Expressive tools: Arbitrary arguments, more manual effort
	\end{itemize}
	\vspace{0.5em}
	\textbf{Other key dimensions:}
	\begin{itemize}
		\item \textbf{Composition support}: Can we reason about pieces separately?
		\item \textbf{Concrete security bounds}: Not just ``secure'' but ``how secure''
		\item \textbf{Link to implementation}: Can guarantees reach real code?
		\item \textbf{Trusted computing base}: Theorem prover? Tool itself? SMT solver?
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		Smaller TCB = Higher confidence
	\end{center}
\end{frame}

\begin{frame}{CryptoVerif: Automated protocol analysis}
	\begin{center}
		\Large
		Best for highly automated protocol analysis
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item Both proof-finding and proof-checking
		\item Excellent for key exchange protocols
		\item Applied $\pi$-calculus input language
	\end{itemize}
	\vspace{0.5em}
	\textbf{Success stories:}
	\begin{itemize}
		\item TLS 1.3 (part of my Ph.D. thesis!)
		\item Signal Protocol (part of my Ph.D. thesis!)
		\item WireGuard VPN
	\end{itemize}
\end{frame}

\begin{frame}{\fstar: Beyond the cryptographic core}
	\begin{center}
		\Large
		Best for analyzing full protocols and systems
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item General-purpose verification language with \textbf{dependent types}
		\item A well-typed program is a \textbf{proof of its specification}
		\item Type checking = Proof checking, with Z3 doing the heavy lifting!
	\end{itemize}
	\vspace{0.5em}
	\textbf{Example:}
	\begin{itemize}
		\item Type: \texttt{point : \{(x,y) : int * int | y*y = x*x*x + a*x + b\}}
		\item This type \textbf{only} accepts valid elliptic curve points
		\item Your code literally cannot process invalid points if it type-checks
	\end{itemize}
\end{frame}

\begin{frame}{\fstar: Real-world impact}
	\textbf{How \fstar handles computational proofs:}
	\begin{enumerate}
		\item Start with detailed protocol description
		\item Transform into ideal program with ideal crypto functionalities
		\item Validate transformation manually, verify final program automatically
	\end{enumerate}
	\vspace{0.5em}
	\textbf{Success stories:}

		\item Discovered new attacks on TLS
		\item miTLS: Verified reference implementation of TLS 1.2
		\item Implemented and verified TLS 1.3 record layer
	\end{itemize}
	\vspace{0.5em}
	\textbf{EasyCrypt successes:} AWS Key Management System, SHA-3 standard
\end{frame}

\begin{frame}{The fine print}
	\begin{center}
		\Large
		Computer-aided security proofs are only as good as the statements being proven
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item Understanding these statements can be \textbf{challenging}
		\item Most proofs rely on \textbf{implicit assumptions}
		\item Reconstructing top-level statements is hard, even for experts
		\item Tools may have bugs (like any software)
	\end{itemize}

\section{Functional Correctness and Efficiency}

\begin{frame}{The fundamental tension}
	\begin{center}
		\Large
		Design-level security is not enough!

		\vspace{0.5em}

		We need implementations that are \textbf{correct}, \textbf{fast}, and \textbf{memory safe}
	\end{center}
	\vspace{1em}
	\begin{columns}[c]
		\begin{column}{0.5\textwidth}
			\textbf{Easy path: Reference implementation}
			\begin{itemize}
				\item \mycheckmark Straightforward correctness
				\item $\times$ Terrible performance
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}
			\textbf{Hard path: Optimize aggressively}
			\begin{itemize}
				\item \mycheckmark Blazing fast
				\item $\times$ Correctness nightmare
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{How far will developers go for speed?}
	\textbf{OpenSSL's approach:}
	\begin{enumerate}
		\item Write Perl code\ldots
		\item \ldots that generates strings of text\ldots
		\item \ldots that other Perl scripts interpret\ldots
		\item \ldots to produce C preprocessor input\ldots
		\item \ldots which generates platform-specific assembly!
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\Large
		This is one of the \textit{fastest} crypto libraries

		\vspace{0.3em}

		\normalsize
		But imagine verifying \textit{that}
	\end{center}
\end{frame}

\begin{frame}{The language dilemma}
	\begin{center}
		\Large
		High-level languages? Too slow.
	\end{center}
	\vspace{0.5em}
	\textbf{Reality of high-performance crypto:}
	\begin{itemize}
		\item C for ``portable'' performance
		\item Assembly for ultimate speed
		\item Platform-specific optimizations everywhere
	\end{itemize}
	\vspace{0.5em}
	\textbf{The cost:}
	\begin{itemize}
		\item Memory safety becomes critical concern
		\item Bugs can leak secrets (Heartbleed)
		\item Verification becomes much harder
	\end{itemize}
	\vspace{0.3em}
	\begin{center}
		\textit{Best-in-class crypto needs manual optimization; compilers can't match expert assembly}
	\end{center}
\end{frame}

\begin{frame}{Even experts make mistakes}
	\begin{center}
		\Large
		A handful of experts write all high-speed crypto
	\end{center}
	\vspace{0.5em}
	\textbf{But even they get it wrong:}
	\begin{itemize}
		\item OpenSSL AES-GCM optimization \rightarrow\ message forgeries
		\item OpenSSL arithmetic bug \rightarrow\ full key recovery
		\item 24 vulnerabilities in libcrypto (2016--2019)
		\item Heartbleed: Memory safety bug leaked secrets
	\end{itemize}
	\vspace{0.5em}
	\textbf{Traditional QA is inadequate:}
	\begin{itemize}
		\item Auditing: Expensive, can miss subtle bugs, doesn't scale
		\item Testing: Cannot cover the huge cryptographic input space
	\end{itemize}
\end{frame}

\begin{frame}{Enter program verification}
	\begin{center}
		\Large
		Cryptographic code is an \textbf{ideal target} for verification
	\end{center}
	\vspace{0.5em}
	\textbf{Why?} Critically important, relatively small codebase, well-defined specs
	\vspace{0.5em}

	\textbf{Two approaches to correctness:}
	\begin{enumerate}
		\item \textbf{Equivalence to reference}: Same outputs for same inputs
		\item \textbf{Functional specification}: Pre/post-conditions on inputs and outputs
	\end{enumerate}
	\vspace{0.5em}
	\textbf{The compiler gap:}
	\begin{itemize}
		\item Most verification happens at source level, but we execute machine code
		\item Solution: Verified compilers (CompCert) or assembly-level verification
	\end{itemize}
\end{frame}

\begin{frame}{Real deployment success stories}
	\begin{columns}[c]
		\begin{column}{0.5\textwidth}
			\textbf{\haclstar in Firefox's NSS:}
			\begin{itemize}
				\item Replaced entire ciphersuite
				\item Performance within 5\% of OpenSSL
				\item Professional packaging
				\item Perfect timing: post-Heartbleed trust crisis
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}
			\textbf{Three critical success factors:}
			\begin{enumerate}
				\item Performance parity
				\item API compatibility (drop-in)
				\item Industry collaboration
			\end{enumerate}
			\vspace{0.5em}
			\textbf{Fiat Crypto}: Powers Google's BoringSSL
		\end{column}
	\end{columns}
	\vspace{0.5em}
	\begin{center}
		\textit{Billions of users rely on verified crypto every day}
	\end{center}
\end{frame}

\begin{frame}{Input languages: Many flavors}
	\textbf{Verification-oriented languages:}
	\begin{itemize}
		\item \textbf{Dafny}: High-level imperative
		\item \textbf{\fstar}: Functional with effects
		\item \textbf{Gallina} (Rocq): Pure functional
	\end{itemize}
	\vspace{0.5em}
	\textbf{Assembly-like languages:}
	\begin{itemize}
		\item \textbf{CryptoLine}: Low-level verification
		\item \textbf{Jasmin}: Assembly with high-level features
		\item \textbf{Vale}: Structured assembly
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		Source-level verification needs a verified compiler for guarantees;\\
		assembly-level verification gives direct machine-code guarantees
	\end{center}
\end{frame}

\begin{frame}{Curve25519: Verified implementations win}
	\textbf{C implementations:}
	\begin{itemize}
		\item \textbf{hacl} (verified) \rightarrow\ Beats donna64 (fast unverified C)
		\item \textbf{fiat} (verified) \rightarrow\ Beats donna64
		\item Multiple verified implementations faster than best unverified C
	\end{itemize}
	\vspace{0.5em}
	\textbf{Assembly implementations:}
	\begin{itemize}
		\item \textbf{evercrypt} (verified) \rightarrow\ Matches fastest unverified
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\Large
		Old wisdom: Verified code = Slow code

		\textbf{New reality: Verified code matches or beats unverified!}

		\normalsize
		\vspace{0.3em}
		No conceptual barrier to verifying the fastest implementations
	\end{center}
\end{frame}

\section{Implementation-Level Security}

\begin{frame}{Side channels: The hidden attack surface}
	\begin{center}
		\Large
		Design-level security assumes attackers only see inputs and outputs.

		\normalsize
		Reality: implementations leak much more!
	\end{center}
	\vspace{0.5em}
	\textbf{Information leaks through:}
	\begin{itemize}
		\item \textbf{Timing}: How long operations take
		\item \textbf{Memory access}: Which cache lines are touched
		\item \textbf{Power consumption}, \textbf{EM radiation}, \textbf{acoustics}
	\end{itemize}
	\vspace{0.5em}
	\textbf{Real attacks:} RSA/AES key recovery, Lucky 13, cross-VM extraction
\end{frame}

\begin{frame}{The constant-time solution}
	\begin{center}
		\Large
		Make execution independent of secrets
	\end{center}
	\vspace{0.5em}
	\textbf{Rules:} No secret-dependent branches, no secret-dependent memory access, no variable-time operations on secrets
	\vspace{0.5em}

	\textbf{Example --- constant-time selection:}
	\begin{flushleft}
		\textit{Insecure:} \texttt{if (secret\_bit) \{ result = a; \} else \{ result = b; \}}
	\end{flushleft}
	\begin{flushleft}
		\textit{Secure:} \texttt{mask = -secret\_bit; result = (a \& mask) | (b \& \textasciitilde mask);}
	\end{flushleft}
	\vspace{0.5em}
	\begin{center}
		Same result, no branches, \textit{much harder to understand}
	\end{center}
\end{frame}

\begin{frame}{Compilers and hardware betray you}
	\textbf{The compiler problem:}
	\begin{itemize}
		\item Compiler ``optimizations'' can reintroduce branches, use lookup tables, short-circuit operations
		\item The compiler doesn't know about your security requirements
	\end{itemize}
	\vspace{0.5em}
	\textbf{The hardware problem:}
	\begin{itemize}
		\item Branch prediction, speculative execution, cache hierarchies
		\item Variable-latency instructions
		\item Spectre/Meltdown-style attacks
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{Even perfect constant-time source code may not survive to execution!}
	\end{center}
\end{frame}

\begin{frame}{Why constant-time is hard}
	\begin{center}
		\Large
		Natural code patterns become forbidden
	\end{center}
	\begin{columns}[t]
		\begin{column}{0.5\textwidth}
			\textbf{You can't write:}
			\begin{enumerate}
				\item \texttt{if (secret\_bit) \{ ... \}}
				\item \texttt{array[secret\_index]}
				\item Return early \texttt{if secret $\stackrel{?}{=}$ 0}
				\item Use division only when \texttt{secret != 0}\footnote{\url{https://kyberslash.cr.yp.to}}
			\end{enumerate}
		\end{column}
		\begin{column}{0.5\textwidth}
			\textbf{Instead, you must:}
			\begin{itemize}
				\item Use bit manipulation tricks
				\item Process all data uniformly
				\item Avoid natural optimizations
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{The TLS patching cycle}
	\begin{center}
		\Large
		Even experts get constant-time wrong
	\end{center}
	\vspace{0.5em}
	\textbf{The vicious cycle:}
	\begin{enumerate}
		\item Original TLS timing vulnerability discovered
		\item Developers patch it
		\item The patch itself creates Lucky 13
		\item Lucky 13 gets patched
		\item The Lucky 13 patch introduces a \textit{new} timing vulnerability!
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{Each ``fix'' introduced new vulnerabilities!}
	\end{center}
\end{frame}

\begin{frame}{Formal leakage models}
	\textbf{Three levels of increasing strictness:}
	\vspace{0.5em}
	\begin{enumerate}
		\item \textbf{Program counter (PC) model}: Leaks control flow only
		      \begin{itemize}
			      \item Protects against simple timing attacks
		      \end{itemize}
		\item \textbf{Constant-time model}: Leaks PC + memory addresses
		      \begin{itemize}
			      \item Protects against cache and timing attacks
			      \item Most common in practice
		      \end{itemize}
		\item \textbf{Size-respecting model}: Leaks PC + memory + operand sizes
		      \begin{itemize}
			      \item Protects against variable-time instruction attacks
		      \end{itemize}
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		Security = Leakage independent of secrets (\textbf{observational non-interference})
	\end{center}
\end{frame}

\begin{frame}{Example: Constant-time selection}
	\textbf{Insecure (natural) code:}
	\begin{flushleft}
		\texttt{if (secret\_bit) \{}\\
		\texttt{~~~~result = a;}\\
		\texttt{\} else \{}\\
		\texttt{~~~~result = b;}\\
		\texttt{\}}
	\end{flushleft}
	\vspace{0.3em}
	\textbf{Secure (constant-time) code:}
	\begin{flushleft}
		\texttt{mask = -secret\_bit;~~// 0x00...0 or 0xFF...F}\\
		\texttt{result = (a \& mask) | (b \& \textasciitilde mask);}
	\end{flushleft}
	\vspace{0.3em}
	\begin{center}
		\textit{Same result, no branches, harder to understand!}
	\end{center}
\end{frame}

\begin{frame}{CAC for implementation security}
	\begin{center}
		\Large
		What if we could \textbf{prove} constant-time?
	\end{center}
	\vspace{0.5em}
	\textbf{CAC tools can:}
	\begin{itemize}
		\item Automatically check constant-time compliance for all possible inputs
		\item Find subtle violations
		\item Even repair non-compliant code
		\item Provide formal guarantees
	\end{itemize}
	\vspace{0.5em}
	\textbf{Important caveats:}
	\begin{itemize}
		\item Proofs are relative to the leakage model
		\item Hardware may leak more than modeled (e.g., Spectre)
		\item Physical attacks bypass all software protections
	\end{itemize}
\end{frame}

\section{Case Study: Lessons from TLS}

\begin{frame}{TLS before 1.3: Reactive patching}
	\begin{center}
		\Large
		One of the most important deployments of cryptography

		\normalsize
		Powers HTTPS, secures billions of connections daily
	\end{center}
	\vspace{0.5em}
	\textbf{The endless loop (pre-1.3):}
	\begin{enumerate}
		\item Attack found \rightarrow\ Quick patch \rightarrow\ Long-term fix \rightarrow\ New attack
	\end{enumerate}
	\vspace{0.3em}
	\textbf{When academics looked at real TLS:} BEAST, CRIME, Lucky 13, POODLE, Logjam\ldots
	\vspace{0.3em}
	\begin{center}
		\textit{No substantial academic analysis during design}
	\end{center}
\end{frame}

\begin{frame}{TLS 1.3: A paradigm shift}
	\begin{center}
		\Large
		From reactive patching to proactive design
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item Academic community consulted \textbf{during} design: 28 drafts over 4+ years
		\item Multiple CAC tools applied: ProVerif, Tamarin, CryptoVerif, \fstar
		\item Real attacks found and fixed \textbf{before} standardization:
		      \begin{itemize}
			      \item Draft 10: Client impersonation in PSK-resumption
			      \item Draft 13: Attack on 0-RTT client authentication
			      \item Draft 21: Unexpected authentication behavior
		      \end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Three lessons from TLS 1.3}
	\textbf{Lesson 1: Formal methods find real flaws}
	\begin{itemize}
		\item Just \textit{trying} to write formal specs uncovers ambiguities and corner cases
		\item \textit{``If you can't specify it clearly, it's probably broken''}
	\end{itemize}
	\vspace{0.3em}
	\textbf{Lesson 2: Protocols are moving targets}
	\begin{itemize}
		\item 28 drafts = 28 different protocols to verify
		\item Machine-checked proofs can evolve with the protocol; manual proofs can't keep pace
	\end{itemize}
	\vspace{0.3em}
	\textbf{Lesson 3: Design for verifiability}
	\begin{itemize}
		\item Small changes (key separation, consistent tagging) dramatically simplify analysis
		\item Negligible performance impact, huge verification benefit
	\end{itemize}
\end{frame}

\begin{frame}{Impact beyond TLS}
	\begin{center}
		\Large
		TLS 1.3 proved formal methods can scale
	\end{center}
	\vspace{0.5em}
	\textbf{The new playbook for protocol design:}
	\begin{enumerate}
		\item Involve formal methods experts early
		\item Design with verification in mind
		\item Use tools throughout development
		\item Update proofs with each draft
		\item Fix issues before standardization
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		From ``ship and patch'' to ``verify and ship''

		\vspace{0.3em}

		\textit{This is the future of secure protocol design}
	\end{center}
\end{frame}

\section{The Verification Boundary}

\begin{frame}{What is a verification boundary?}
	\begin{center}
		\Large
		Every formally verified system embeds a \textbf{verification boundary}
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item The interface between code with \textbf{machine-checked proofs} and code that is \textbf{trusted without them}
		\item Outside the boundary: compilers, OS, hardware, platform intrinsics, ``glue code''
		\item The existence of this boundary is well understood\ldots
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{But it is not always communicated clearly}
	\end{center}
\end{frame}

\begin{frame}{Good boundary documentation exists}
	\begin{itemize}
		\item \textbf{CompCert}: Explicitly documents which compiler passes are verified and which (parser, assembler, linker) are not
		\item \textbf{seL4}: Maintains a dedicated page listing every trust assumption about hardware behavior
		\item \textbf{EverCrypt}: Explicitly identifies compiler, Vale embedding semantics, and hardware as trusted components
		\item \textbf{Jasmin}: Documents which compiler passes have machine-checked proofs
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\Large
		These projects treat verification boundaries as \textbf{first-class documentation}
	\end{center}
\end{frame}

\begin{frame}{When boundaries go wrong}
	\begin{center}
		\Large
		What happens when a library is marketed as ``formally verified'' \textbf{without qualification}?
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item Users reasonably interpret this as a comprehensive guarantee
		\item The gap between verification \textit{claims} and verification \textit{reality} creates false assurance
		\item We call this gap \textbf{``verification theatre''}
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{Deploying formal methods with unqualified claims creates an appearance of correctness while leaving significant code paths unverified}
	\end{center}
\end{frame}

\begin{frame}{Case study: Cryspen's libcrux}
	\begin{itemize}
		\item Cryspen: ``services and software for high assurance software''
		\item libcrux repository: ``the formally verified crypto library''

\begin{frame}{The formal verification pipeline}
	\begin{center}
		\Large
		How does Cryspen's verification work?
	\end{center}
	\vspace{0.5em}
	\begin{enumerate}
		\item \textbf{Specification}: Algorithm spec written in hacspec (Rust subset) or \fstar
		\item \textbf{Implementation}: Optimized Rust with multiple backends (portable, AVX2, NEON)
		\item \textbf{Extraction and Proof}: hax toolchain extracts Rust to \fstar; provers verify correctness, safety, and secret independence
		\item \textbf{Compilation}: Verified source compiled by \textbf{unverified} compiler (LLVM/rustc)
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{Each stage introduces a trust boundary}
	\end{center}
\end{frame}

\section{Type I: Unverified Platform Abstractions}

\begin{frame}{Type I: Unverified Platform Abstractions}
	\begin{center}
		\Large
		The verified code depends on platform intrinsics treated as \textbf{axioms}
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item Formal proofs \textbf{assume} these operations implement their specified semantics
		\item The actual implementations may contain bugs
		\item Marked \texttt{\#[hax\_lib::opaque]}: verifier accepts specification without proof
	\end{itemize}
\end{frame}

\begin{frame}{V1: Incorrect ARM64 SHA-3 intrinsic}
	\begin{itemize}
		\item The \texttt{\_vxarq\_u64} intrinsic wrapper for ARM64 SHA-3
		\item On platforms \textbf{without} SHA-3 hardware support, a fallback path is used
		\item The fallback implementation was \textbf{incorrect}
		\item Produced wrong cryptographic output on certain ARM64 devices
	\end{itemize}
	\vspace{0.5em}
	\textbf{Why verification missed it:}
	\begin{itemize}
		\item Function marked as opaque to the verifier
		\item Verifier accepted the \textit{specification} as an axiom
		\item The axiom was correct; the \textit{implementation} was wrong
		\item Discovered independently by Filippo Valsorda
	\end{itemize}
\end{frame}

\begin{frame}{V2: Cross-backend endianness bug}
	\begin{itemize}
		\item ML-KEM incremental API: \texttt{from\_bytes}/\texttt{to\_bytes} serialization
		\item Portable backend used \textbf{different byte order} than SIMD backends
		\item Serialize state on one platform, deserialize on another \rightarrow\ \textbf{failure}
	\end{itemize}
	\vspace{0.5em}
	\textbf{Real-world impact:}
	\begin{itemize}
		\item Caused \textbf{decryption failures in Signal's post-quantum ratchet}
		\item Reported by GuuJiang (December 2025)
		\item Each backend was individually correct, but they were \textbf{mutually incompatible}
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{Verification proved each backend correct in isolation; the cross-backend property was never specified}
	\end{center}
\end{frame}

\begin{frame}{Type I: Lessons}
	\begin{center}
		\Large
		What Type I failures teach us
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item Platform abstractions are \textbf{axioms}: the verifier takes them on faith
		\item Cross-platform properties (endianness, representation) are rarely specified
		\item Testing each backend individually is \textbf{not enough}
		\item Need cross-backend compatibility testing
		\item Axiomatized specifications should be \textbf{proved or exhaustively tested} against a reference
	\end{itemize}
\end{frame}

\section{Type II: Unverified Protocol Logic}

\begin{frame}{Type II: Unverified Protocol Logic}
	\begin{center}
		\Large
		Verified primitives used correctly\ldots

		but the protocol logic around them has bugs
	\end{center}
	\vspace{1em}
	\textbf{V3: Missing X25519 zero check}
	\begin{itemize}
		\item RFC 9180 \textbf{requires} checking if DH output is all-zero
		\item hpke-rs returns the raw result without validation
		\item Small-subgroup attack: adversary can force shared secret to zero
	\end{itemize}
	\vspace{0.5em}
	\textbf{V4: Nonce reuse via integer overflow}
	\begin{itemize}
		\item HPKE sequence counter is \texttt{u32} (wraps at $2^{32}$)
		\item In release builds, wrapping is \textbf{silent}: counter goes back to 0
		\item After $2^{32}$ messages: \textbf{nonce reuse!}
	\end{itemize}
\end{frame}

\begin{frame}{Type II: Lessons}
	\begin{center}
		\Large
		What Type II failures teach us
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item Verifying the \textbf{crypto primitive} does not verify the \textbf{protocol}
		\item Standard-mandated checks (RFC 9180 zero check) live in protocol logic
		\item Integer overflow behavior depends on build mode (debug vs.\ release)
		\item Protocol-level code needs its own analysis---symbolic or otherwise
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{The most dangerous bugs live in the space \textbf{between} verified components}
	\end{center}
\end{frame}

\section{Type III: Specification Incompleteness}

\begin{frame}{Type III: Specification Incompleteness}
	\begin{center}
		\Large
		Code is \textbf{provably correct} per its spec\ldots

		but the spec doesn't capture all real-world requirements
	\end{center}
	\vspace{1em}
	\textbf{V5: ECDSA missing low-$S$ normalization}
	\begin{itemize}
		\item ECDSA signatures have a malleability: both $(r, s)$ and $(r, n-s)$ are valid
		\item Bitcoin, Ethereum, and other systems \textbf{require} $s \leq n/2$ (``low-$S$'')
		\item libcrux's \fstar specification doesn't include this requirement
		\item Implementation is provably correct per spec, but \textbf{rejected by practical systems}
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{The spec was incomplete, not the proof}
	\end{center}
\end{frame}

\section{Type IV: Unverified Wrapper Logic}

\begin{frame}{Type IV: Unverified Wrapper Logic}
	\begin{center}
		\Large
		The verified core is correct, but the \textbf{API layer} introduces defects
	\end{center}
	\vspace{1em}
	\textbf{V6: Ed25519 double clamping}
	\begin{itemize}
		\item Key generation wrapper applies clamping \textbf{before} hashing
		\item Should be: hash first, then clamp the result
		\item Reduces entropy of generated keys
	\end{itemize}
	\vspace{0.5em}
	\textbf{V7: AES-GCM decryption panic}
	\begin{itemize}
		\item Decryption wrapper calls \texttt{.unwrap()} on the result
		\item If decryption fails (e.g., tampered ciphertext): \textbf{program panics}
		\item Denial-of-service via any invalid ciphertext
		\item The correct error handling was \textit{commented out} in the source
	\end{itemize}
\end{frame}

\begin{frame}{V8: ML-DSA norm check rendered dead code}
	\textbf{FIPS 204 requirement:} Reject signature if signer response exceeds $\gamma_1 - \beta$
	\vspace{0.5em}

	\textbf{The bug:}
	\begin{flushleft}
		\texttt{if vector\_infinity\_norm\_exceeds(response, \textbf{(2 << GAMMA1\_EXPONENT)} - BETA)}
	\end{flushleft}
	\vspace{0.3em}
	\begin{itemize}
		\item \texttt{2 << GAMMA1\_EXPONENT} computes $2\gamma_1$, not $\gamma_1$
		\item The bound is \textbf{doubled}: $2\gamma_1 - \beta$ instead of $\gamma_1 - \beta$
		\item Since the signer response is already bounded by $\gamma_1 - \beta$, the check \textbf{always passes}
		\item A FIPS 204 security check is effectively \textbf{dead code}
	\end{itemize}
	\vspace{0.3em}
	\begin{center}
		\textit{libcrux accepts non-conforming ML-DSA signatures}
	\end{center}
\end{frame}

\begin{frame}{V9: ML-DSA hint deserialization off-by-one}
	\textbf{FIPS 204 requirement:} Check that hint count does not exceed maximum
	\vspace{0.5em}

	\textbf{The bug:} The overflow check compares against the \textbf{previous} iteration's count, not the current one
	\begin{flushleft}
		\texttt{if (\textbf{previous\_true\_hints\_seen} > max\_ones\_in\_hint) \{}\\
		\texttt{~~~~malformed\_hint = true;}\\
		\texttt{\}}
	\end{flushleft}
	\begin{itemize}
		\item Should check \texttt{current\_true\_hints\_seen > max\_ones\_in\_hint}
		\item The final iteration's bound is \textbf{never validated}
		\item Allows non-conforming signatures to pass verification
	\end{itemize}
\end{frame}

\begin{frame}{Types I--IV: Summary}
	\begin{center}
		\Large
		Nine vulnerabilities outside the verified core
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item \textbf{Real-world impact}: Signal PQ ratchet failures (V2), potential DoS (V7), non-conforming ML-DSA signatures (V8, V9)
		\item \textbf{Common pattern}: The verified primitive works correctly, but everything \textit{around} it is unverified
		\item \textbf{Key insight}: Formal verification of primitives provides a \textbf{false sense of security} when the surrounding code is not analyzed
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{But the most surprising findings are still ahead\ldots}
	\end{center}
\end{frame}

\section{Type V: Specification Errors and False Proofs}

\begin{frame}{Type V: The most insidious failures}
	\begin{center}
		\Large
		Bugs \textbf{inside} the formally verified code itself
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item The formal \textbf{specification} is wrong
		\item Or the \textbf{proofs} contain false claims silently accepted by the build system
		\item These undermine the \textbf{entire foundation} on which all verification rests
		\item Even complete proofs only establish correctness with respect to the \textbf{wrong definition}
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{Four concrete examples in ML-KEM and ML-DSA\ldots}
	\end{center}
\end{frame}

\begin{frame}{How can bugs exist inside ``verified'' code?}
	\begin{center}
		\Large
		Three mechanisms that allow false proofs to survive
	\end{center}
	\vspace{0.5em}
	\begin{enumerate}
		\item \textbf{Wrong specifications}: The formal spec is a mathematical statement---if the statement is wrong, proving it correct doesn't help
		\item \textbf{Admitted proofs}: The build system can skip proof checking entirely while reporting success
		\item \textbf{False axioms}: Unproved assumptions can contradict reality; all downstream proofs become unsound
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{Let's see each mechanism in action\ldots}
	\end{center}
\end{frame}

\begin{frame}{V10: Wrong decompression constant}
	\textbf{FIPS 203 specification:}
	\begin{flushleft}
		$\mathtt{decompress}_d(x) = \lceil x \cdot q / 2^d \rfloor = (x \cdot 3329 + 2^{d-1}) / 2^d$
	\end{flushleft}
	\vspace{0.3em}
	\textbf{libcrux \fstar specification:}
	\begin{flushleft}
		\texttt{let decompress\_d d x = (x * v\_FIELD\_MODULUS + \textbf{1664}) / pow2 d}
	\end{flushleft}
	\vspace{0.5em}
	\begin{itemize}
		\item $1664 = (q-1)/2 = (3329-1)/2$, a fixed value unrelated to $d$
		\item The correct rounding constant should be $2^{d-1}$, which varies with $d$
		\item ML-KEM uses $d \in \{1, 4, 5, 10, 11\}$: all five values produce wrong results
	\end{itemize}
	\vspace{0.3em}
	\begin{center}
		\textit{The specification disagrees with the standard it claims to formalize}
	\end{center}
\end{frame}

\begin{frame}{V11: Missing inverse NTT}
	\textbf{ML-KEM encryption computes:}
	\begin{itemize}
		\item $\mathbf{u} = \text{NTT}^{-1}(\hat{A}^T \circ \hat{r}) + e_1$ \quad (correct in spec)
		\item $v = \text{NTT}^{-1}(\hat{t}^T \circ \hat{r}) + e_2 + \lceil q/2 \rfloor \cdot m$ \quad (FIPS 203)
	\end{itemize}
	\vspace{0.5em}
	\textbf{libcrux \fstar specification for $v$:}
	\begin{flushleft}
		\texttt{let v = poly\_add (poly\_add (\textbf{vector\_dot\_product\_ntt} t\_as\_ntt r\_as\_ntt) error\_2) mu}
	\end{flushleft}
	\vspace{0.3em}
	\begin{itemize}
		\item The result of \texttt{vector\_dot\_product\_ntt} is in the \textbf{NTT domain}
		\item But \texttt{error\_2} and \texttt{mu} are in the \textbf{coefficient domain}
		\item Adding them together is mathematically \textbf{nonsensical}
		\item The missing \texttt{poly\_inv\_ntt} call makes the spec wrong
		\item Yet it \textbf{type-checks} because both are just polynomial types
	\end{itemize}
\end{frame}

\begin{frame}{V12: A false serialization proof}
	\textbf{The claim:}
	\begin{flushleft}
		\texttt{let deserialize\_12\_bit\_vec\_lemma\_bounded (v: t\_Array u8 (sz 24))}\\
		\texttt{~~: squash (forall (i: nat \{i < 16\}).}\\
		\texttt{~~~~~~bounded (Seq.index result.f\_elements i) \textbf{1})}
	\end{flushleft}
	\vspace{0.5em}
	\begin{itemize}
		\item This claims that 12-bit deserialized values are bounded by $2^{\textbf{1}} = 2$
		\item The correct bound should be $2^{12} = 4096$
		\item This is \textbf{obviously false}: 12-bit values can be up to 4095
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\Large
		How does a false proof get accepted?
	\end{center}
\end{frame}

\begin{frame}{The lax-mode escape hatch}
	\textbf{The proof tactic:}
	\begin{flushleft}
		\texttt{let prove\_bit\_vector\_equality' () : Tac unit =}\\
		\texttt{~~if \textbf{lax\_on ()}}\\
		\texttt{~~then \textbf{iterAll tadmit}}\\
		\texttt{~~else prove\_bit\_vector\_equality'' ()}
	\end{flushleft}
	\vspace{0.5em}
	\textbf{Three-layer failure mechanism:}
	\begin{enumerate}
		\item The Makefile \textbf{admits this module by default} (\texttt{ADMIT\_MODULES})
		\item When admitted, \texttt{lax\_on()} returns true
		\item The tactic calls \texttt{tadmit}: \textbf{all proof goals are accepted unconditionally}
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{A false proof becomes a ``verified'' proof whenever the module is admitted}

		\textit{Full verification is \textbf{never triggered} in any automated pipeline}
	\end{center}
\end{frame}

\begin{frame}{V13: Wrong multiplication specification}
	\textbf{ML-DSA AVX2 intrinsics specification:}
	\begin{flushleft}
		\texttt{let i16\_mul\_32extended (x y: i16): i32 =}\\
		\texttt{~~(cast x : i32) *! (cast \textbf{x} : i32) ~~// should be: cast \textbf{y}}
	\end{flushleft}
	\vspace{0.5em}
	\begin{itemize}
		\item The second parameter \texttt{y} is \textbf{completely ignored}
		\item The body computes $x^2$ instead of $x \cdot y$
		\item This specification is \textbf{axiomatized}: marked \texttt{[@@ "opaque\_to\_smt"]}
		\item The axiom is \textbf{never proved} against a reference model
		\item All proofs building on this axiom are \textbf{unsound}
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{The verified ML-DSA AVX2 proofs prove properties about $x^2$, not $x \cdot y$}
	\end{center}
\end{frame}

\begin{frame}{Why Type V bugs survive}
	\begin{center}
		\Large
		Why do specification errors and false proofs persist?
	\end{center}
	\vspace{0.5em}
	\begin{enumerate}
		\item \textbf{Build system admits proofs by default}: \texttt{ADMIT\_MODULES} in Makefile
		\item \textbf{Lax-mode escape hatches}: Tactics silently accept all claims
		\item \textbf{Axioms are never proved}: \texttt{val} declarations without implementations
		\item \textbf{Full verification never triggered}: Not in CI, not on any automated pipeline
		\item \textbf{Manual status documents}: Can claim verification that doesn't exist
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		Specifications can contain mathematical errors, proofs can make false claims, axioms can contradict their definitions---\textbf{all simultaneously}
	\end{center}
\end{frame}

\begin{frame}{Type V: Summary}
	\begin{center}
		\Large
		Four bugs inside the ``verified'' code
	\end{center}
	\vspace{0.5em}
	\begin{center}
		\begin{tabular}{|l|l|l|}
			\hline
			\textbf{Vuln} & \textbf{Description}         & \textbf{Algorithm} \\
			\hline
			V10           & Wrong rounding constant      & ML-KEM             \\
			\hline
			V11           & Missing inverse NTT          & ML-KEM             \\
			\hline
			V12           & False serialization bound    & ML-KEM             \\
			\hline
			V13           & $x^2$ instead of $x \cdot y$ & ML-DSA             \\
			\hline
		\end{tabular}
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item The deployed Rust implementations are currently correct---the risk is \textbf{indirect}
		\item A developer who trusts the spec and refactors to match it would silently introduce the errors
		\item The proofs are \textbf{not checked in any CI pipeline}
	\end{itemize}
\end{frame}

\section{Systemic Issues}

\begin{frame}{The ADMIT\_MODULES pattern}
	\textbf{From the ML-KEM extraction Makefile:}
	\begin{flushleft}
		\texttt{ADMIT\_MODULES = Libcrux\_ml\_kem.Ind\_cpa.fst \textbackslash}\\
		\texttt{~~~~Libcrux\_ml\_kem.Sampling.fst \textbackslash}\\
		\texttt{~~~~Libcrux\_ml\_kem.Vector.Neon.Arithmetic.fst \textbackslash}\\
		\texttt{~~~~Libcrux\_ml\_kem.Vector.Neon.Compress.fst \textbackslash}\\
		\texttt{~~~~...}
	\end{flushleft}
	\vspace{0.5em}
	\begin{itemize}
		\item These modules are \textbf{admitted by default}: proofs are not checked
		\item ``Lax checking'' means the \fstar type-checker accepts the code, but proofs are \textbf{not verified}
		\item Like spell-checking a math proof: syntax is fine, content may be wrong
	\end{itemize}
\end{frame}

\begin{frame}{Admitted proofs as systemic risk}
	\begin{center}
		\Large
		Admitted proofs are \textbf{more dangerous} than unverified code
	\end{center}
	\vspace{0.5em}
	\textbf{Unverified code (V1--V9):}
	\begin{itemize}
		\item Was never claimed to be verified
		\item Developers and users can identify it as a risk
	\end{itemize}
	\vspace{0.5em}
	\textbf{Admitted proofs (V10--V13):}
	\begin{itemize}
		\item Code \textit{is} claimed to be verified
		\item But the proofs are \textbf{silently not checked}
		\item Creates false confidence in the verified core
		\item A developer who trusts the spec and refactors the implementation to match it would \textbf{introduce the spec's errors into production}
	\end{itemize}
\end{frame}

\begin{frame}{What ``lax checking'' actually means}
	\begin{center}
		\begin{tabular}{|l|c|c|c|}
			\hline
			\textbf{Level}    & \textbf{Syntax} & \textbf{Types} & \textbf{Proofs} \\
			\hline
			Lax checking      & $\checkmark$    & $\checkmark$   & --              \\
			\hline
			Runtime safety    & $\checkmark$    & $\checkmark$   & Partial         \\
			\hline
			Full verification & $\checkmark$    & $\checkmark$   & $\checkmark$    \\
			\hline
		\end{tabular}
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item Lax checking confirms the code is \textbf{well-formed} \fstar
		\item It does \textbf{not} confirm that the proofs are valid
		\item A false lemma claiming $2^{12} < 2^1$ passes lax checking
		\item The NEON backend has \textbf{only} lax checking for all modules
	\end{itemize}
\end{frame}

\section{Comparative Analysis}

\begin{frame}{How AWS does it differently}
	\begin{center}
		\Large
		The same failure types exist in AWS's verified libcrypto---but they are managed differently
	\end{center}
	\vspace{0.5em}
	\textbf{AWS verification projects:}
	\begin{itemize}
		\item aws-lc-verification (SAW/Cryptol)
		\item s2n-bignum (HOL Light)
		\item mlkem-native (CBMC)
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		Same verification boundary challenges, \textbf{very different practices}
	\end{center}
\end{frame}

\begin{frame}{What good boundary management looks like}
	\begin{center}
		\begin{tabular}{|l|c|c|}
			\hline
			\textbf{Practice}               & \textbf{Cryspen} & \textbf{AWS} \\
			\hline
			Soundness documents             & Manual/stale     & Structured   \\
			\hline
			Proofs in CI                    & Not triggered    & Every commit \\
			\hline
			Admitted modules documented     & In Makefile only & Explicit     \\
			\hline
			Verification scope communicated & Unqualified      & Qualified    \\
			\hline
			Axioms tested/proved            & Neither          & Tested       \\
			\hline
		\end{tabular}
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item AWS's soundness documents enumerate: verified properties, assumptions, residual risks, and open gaps
		\item The gold standard remains \textbf{seL4}: dedicated page listing every trust assumption
	\end{itemize}
\end{frame}

\begin{frame}{Parallels with CompCert}
	\begin{center}
		\Large
		Yang et al.'s Csmith campaign (2011)
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item Randomly generated C programs to find compiler bugs
		\item \textbf{Found zero wrong-code bugs} in CompCert's verified middle-end
		\item \textbf{Did find bugs} in the unverified front-end
	\end{itemize}
	\vspace{0.5em}
	\textbf{Direct parallel to our findings:}
	\begin{itemize}
		\item libcrux's verified core is more robust than its wrappers
		\item But we go further: the verified core can contain \textbf{specification errors} (V10, V11), \textbf{false proofs} (V12), and \textbf{false axioms} (V13) when the build system doesn't check them
	\end{itemize}
\end{frame}

\begin{frame}{Thompson's ``Reflections on Trusting Trust''}
	\begin{center}
		\Large
		``Who verifies the verifiers?''
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item Ken Thompson (1984): You can't trust code you didn't write yourself
		\item Remains relevant: proofs dispatched through \textbf{unverified SMT solvers}
		\item \fstar relies on Z3 for proof obligations
		\item Z3 is a complex piece of software with its own bugs
		\item The trusted computing base is \textbf{never zero}
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{But we can at least ensure the proofs are \textbf{actually run}}
	\end{center}
\end{frame}

\section{Recommendations and Lessons}

\begin{frame}{Five recommendations}
	\begin{enumerate}
		\item \textbf{Machine-readable verification manifests}: Every release should include a manifest of which code paths are verified, against which properties, for which platforms. Generated from build system, not manually maintained.
		\item \textbf{Proofs in CI on every commit}: All proof obligations discharged in CI. Too-slow proofs flagged as incomplete, not silently admitted.
		\item \textbf{No lax-mode escape hatches}: Proof tactics should fail loudly in lax mode, not silently admit all goals.
	\end{enumerate}
\end{frame}

\begin{frame}{Five recommendations (continued)}
	\begin{enumerate}
		\setcounter{enumi}{3}
		\item \textbf{Clear scope communication}: Claims of ``formally verified'' should specify which properties, which code paths, which platforms. Unqualified claims should be reserved for systems with minimal TCB (CompCert, seL4 level).
		\item \textbf{Defense in depth}: Formal verification is one component of a defense-in-depth strategy. Code review, specification compliance auditing, cross-platform testing, and fuzzing catch categories of bugs that formal verification does not.
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{Traditional practices remain essential alongside formal methods}
	\end{center}
\end{frame}

\begin{frame}{Disclosure: A cautionary tale}
	\begin{itemize}
		\item Feb 5, 2026: We submit PRs with tested fixes for V3--V6
		\item Feb 5--6: Cryspen \textbf{blocks our GitHub account}, closes all PRs
		\item Feb 9: Cryspen merges our fixes \textbf{without attribution}
		\item Feb 12: Cryspen publishes response stating ``no bugs have been found in the verified code''
		\item Feb 17: We disclose V8--V9 (FIPS 204 violations)
		\item Mar 4: Cryspen merges fixes for V8--V9, crediting the discoverer
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{``No bugs in the verified code'' is directly contradicted by V10--V13}
	\end{center}
\end{frame}

\begin{frame}{Why this matters: The trust chain}
	\begin{center}
		\Large
		Users trust the library \rightarrow\ Library trusts the proofs \rightarrow\ Proofs trust the spec
	\end{center}
	\vspace{1em}
	\begin{itemize}
		\item If the spec is wrong (V10, V11): the entire chain is broken
		\item If the proofs are false (V12): the chain is illusory
		\item If axioms are wrong (V13): the chain is unsound
		\item If proofs are never checked: the chain doesn't exist
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{Trust is only as strong as the weakest link}
	\end{center}
\end{frame}

\begin{frame}{Connecting Part 1 and Part 2}
	\begin{center}
		\Large
		Part 1 showed us the \textbf{promise} of CAC

		\vspace{0.5em}

		Part 2 shows us the \textbf{pitfalls}
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item The same \fstar technology behind \haclstar (success story) powers libcrux
		\item The difference is not the tools---it's the \textbf{engineering practices}
		\item EverCrypt documents its trust assumptions; libcrux does not
		\item \haclstar ran full proofs in CI; libcrux admits modules by default
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{Verification is only as strong as the process around it}
	\end{center}
\end{frame}

\begin{frame}{Auditing methodology}
	\begin{center}
		\Large
		A lightweight methodology for detecting verification boundary failures
	\end{center}
	\vspace{0.5em}
	\begin{enumerate}
		\item \textbf{Map the verification boundary}: Find extraction annotations, build system admission flags, unverified wrappers, platform-specific code paths
		\item \textbf{Audit specifications against standards}: Check constants, domain transformations, required validations against FIPS/RFC
		\item \textbf{Verify proofs are actually checked}: Run the full verifier, not just lax checking; look for \texttt{tadmit} and \texttt{assume}
		\item \textbf{Cross-reference documentation}: Compare claimed verification status against actual build system behavior
	\end{enumerate}
\end{frame}

\begin{frame}{What ``formally verified'' should mean}
	\begin{center}
		\Large
		A spectrum of claims
	\end{center}
	\vspace{0.5em}
	\begin{enumerate}
		\item \textbf{``Lax-checked''}: Syntax and types are valid (minimal guarantee)
		\item \textbf{``Partially verified''}: Some modules have completed proofs, others are admitted
		\item \textbf{``Verified with caveats''}: All proofs checked, but with documented trust assumptions (compiler, hardware, axioms)
		\item \textbf{``Fully verified''}: Minimal TCB, all proofs in CI, explicit boundary documentation (CompCert/seL4 level)
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{Most projects are at level 2 or 3. Claims should reflect this.}
	\end{center}
\end{frame}

\begin{frame}{Verification theatre, defined}
	\begin{center}
		\Large
		\textbf{Verification theatre} is the practice of deploying formal methods in a manner that:
	\end{center}
	\vspace{0.5em}
	\begin{enumerate}
		\item Establishes machine-checked proofs for a \textbf{subset} of code and properties, where some proofs may themselves be \textbf{unchecked, vacuous, or built on false axioms}
		\item Communicates these proofs using language that implies \textbf{comprehensive coverage}
		\item Fails to adequately disclose the verification boundary, unverified components, unchecked proofs, or properties not covered
	\end{enumerate}
	\vspace{0.5em}
	\begin{center}
		\textit{Echoes Schneier's ``security theater'': an appearance of security that doesn't match reality}
	\end{center}
\end{frame}

\begin{frame}{Looking forward: Post-quantum transition}
	\begin{center}
		\Large
		The stakes are rising
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item ML-KEM and ML-DSA are the \textbf{new NIST standards} for post-quantum cryptography
		\item They will be deployed in virtually every secure communication system
		\item Getting their implementations right is \textbf{critical}
		\item libcrux provides the ML-KEM implementation used by Google
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\textit{The post-quantum transition is the moment when verification boundary practices matter most}
	\end{center}
\end{frame}

\begin{frame}{The bottom line}
	\begin{center}
		\Large
		Formal verification is genuinely valuable
	\end{center}
	\vspace{0.5em}
	\begin{itemize}
		\item Machine-checked proofs of correctness, memory safety, and secret independence are real contributions to software security
		\item \haclstar, Fiat-Crypto, Jasmin, EverCrypt: these are genuine achievements
	\end{itemize}
	\vspace{0.5em}
	\begin{center}
		\Large
		But partial verification $\neq$ comprehensive assurance
	\end{center}
	\vspace{0.3em}
	\begin{itemize}
		\item Treat verification boundaries as first-class documentation
		\item Ensure build systems actually verify the proofs they claim to check
		\item Formal methods \textbf{complement}, rather than replace, engineering practices
	\end{itemize}
	\vspace{0.3em}
	\begin{center}
		\textbf{Verify the verifiers.}
	\end{center}
\end{frame}
